```typescript
rpcHandlerManager.registerHandler('resume-session', async (params: { sessionId: string }) => {
    const { sessionId } = params;
    if (!sessionId) {
        return { ok: false, error: 'Session ID is required' };
    }

    // Look up the Claude session ID from persisted state
    const daemonState = readDaemonState();
    const sessionInfo = daemonState?.activeSessions?.find(s => s.idleSessionId === sessionId);
    const claudeSessionId = sessionInfo?.claudeSessionId;

    if (!claudeSessionId) {
        return { ok: false, error: 'No Claude session ID found for this session' };
    }

    const result = await spawnSession({
        sessionId: claudeSessionId,
        directory: sessionInfo.workingDirectory || process.cwd(),
        startedBy: 'app-resume',
    });

    return result;
});
```

**Step 1:** Create `pushService.ts`:

```typescript
import { db } from '../db';

interface PushPayload {
    to: string;
    title: string;
    body: string;
    badge?: number;
    sound?: 'default';
    data?: Record<string, unknown>;
}

export async function sendPushToAccount(accountId: string, title: string, body: string) {
    const tokens = await db.accountPushToken.findMany({
        where: { accountId },
        select: { token: true },
    });

    if (tokens.length === 0) return;

    const messages: PushPayload[] = tokens.map(t => ({
        to: t.token,
        title,
        body: body.length > 100 ? body.substring(0, 97) + '...' : body,
        sound: 'default',
    }));

    const response = await fetch('https://exp.host/--/api/v2/push/send', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(messages),
    });

    // Handle invalid tokens (remove from DB)
    if (response.ok) {
        const result = await response.json();
        // Check for DeviceNotRegistered errors and remove stale tokens
    }
}
```
