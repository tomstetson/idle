### Task 4: Regression Tests — PGlite Bytes Serialization

**Files:**
- Create: `packages/idle-server/sources/app/api/routes/sessionRoutes.regression.test.ts`
- Create: `packages/idle-server/sources/app/api/routes/machinesRoutes.regression.test.ts`

**Step 1: Write the session regression test**

This test verifies that `dataEncryptionKey` (a Prisma `Bytes` field) is stored and retrieved correctly as a Buffer, not as a JSON object. The bug was: `new Uint8Array(Buffer.from(key, 'base64'))` caused PGlite to serialize as `{"0":0,"1":152,...}` instead of binary.

Test file: `packages/idle-server/sources/app/api/routes/sessionRoutes.regression.test.ts`

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest';

/**
 * Regression test for PGlite Bytes field serialization.
 *
 * Bug: Wrapping Buffer.from() in new Uint8Array() caused PGlite's Prisma
 * adapter to serialize Bytes fields as JSON objects {"0":0,"1":152,...}
 * instead of binary data. This broke session creation with 500 errors.
 *
 * Fix: Use Buffer.from(value, 'base64') directly — no Uint8Array wrapper.
 * See commit 38046952.
 */
describe('PGlite Bytes field regression', () => {
    it('Buffer.from returns a Buffer instance, not a plain Uint8Array', () => {
        const testKey = 'dGVzdC1lbmNyeXB0aW9uLWtleS0xMjM0NTY3OA=='; // base64
        const result = Buffer.from(testKey, 'base64');

        // Buffer IS a Uint8Array subclass, but Prisma/PGlite handles it correctly
        expect(Buffer.isBuffer(result)).toBe(true);
        expect(result).toBeInstanceOf(Uint8Array);
        expect(result).toBeInstanceOf(Buffer);
    });

    it('new Uint8Array(Buffer.from(...)) strips Buffer identity — this was the bug', () => {
        const testKey = 'dGVzdC1lbmNyeXB0aW9uLWtleS0xMjM0NTY3OA==';
        const badResult = new Uint8Array(Buffer.from(testKey, 'base64'));

        // This is a plain Uint8Array, NOT a Buffer — PGlite serializes it wrong
        expect(Buffer.isBuffer(badResult)).toBe(false);
        expect(badResult).toBeInstanceOf(Uint8Array);
        expect(badResult).not.toBeInstanceOf(Buffer);
    });

    it('Buffer round-trips through base64 correctly', () => {
        const original = 'test-encryption-key-12345678';
        const base64 = Buffer.from(original).toString('base64');
        const restored = Buffer.from(base64, 'base64');

        expect(restored.toString()).toBe(original);
        expect(Buffer.isBuffer(restored)).toBe(true);
    });
});
```

```typescript
/**
 * Regression test for auth token persistence.
 *
 * Bug: tokenStorage used sessionStorage on web, which clears when the
 * tab closes. Users had to re-authenticate every time they closed the browser.
 *
 * Fix: Switch to localStorage for persistent auth across browser sessions.

describe('tokenStorage web persistence', () => {
    // We test the core principle: localStorage persists, sessionStorage doesn't
    // The actual TokenStorage module has Platform.OS checks that are hard to mock
    // in a Node test environment, so we test the contract directly.

    it('localStorage persists data across simulated tab close', () => {
        // localStorage is available in vitest jsdom, but we're in node env
        // Test the principle: the key difference between the two storage APIs
        const mockLocalStorage = new Map<string, string>();
        const mockSessionStorage = new Map<string, string>();

        // Simulate storing credentials
        const creds = JSON.stringify({ token: 'test-token', secret: 'test-secret' });
        mockLocalStorage.set('idle-auth-credentials', creds);
        mockSessionStorage.set('idle-auth-credentials', creds);

        // Simulate tab close: sessionStorage clears, localStorage persists
        mockSessionStorage.clear();

        expect(mockLocalStorage.get('idle-auth-credentials')).toBe(creds);
        expect(mockSessionStorage.get('idle-auth-credentials')).toBeUndefined();
```

**Step 1: Create auth helper**

Handles creating ephemeral accounts and generating auth credentials programmatically (no browser needed for server-side auth).

```typescript
// packages/idle-e2e/helpers/auth.ts
import tweetnacl from 'tweetnacl';
import axios from 'axios';

const SERVER_URL = process.env.IDLE_SERVER_URL || 'https://idle-api.northglass.io';

export interface TestAccount {
    publicKey: Uint8Array;
    secretKey: Uint8Array;
    token: string;
}

/**
 * Create a fresh test account by completing the full auth flow programmatically.
 * This simulates what happens when CLI + web app do the auth handshake.
 */
export async function createTestAccount(): Promise<TestAccount> {
    // Generate ephemeral keypair (simulates CLI side)
    const terminalKeypair = tweetnacl.box.keyPair();

    // Step 1: CLI creates auth request
    await axios.post(`${SERVER_URL}/v1/auth/request`, {
        publicKey: Buffer.from(terminalKeypair.publicKey).toString('base64'),
        supportsV2: true,
    });

    // Step 2: Create an account via direct auth (simulates having an existing account)
    // For E2E tests, we use challenge-response auth to create a fresh account
    const accountKeypair = tweetnacl.sign.keyPair();
    const challenge = Buffer.from(tweetnacl.randomBytes(32)).toString('base64');
    const signature = Buffer.from(
        tweetnacl.sign.detached(
            Buffer.from(challenge, 'base64'),
            accountKeypair.secretKey
        )
    ).toString('base64');

    const authResponse = await axios.post(`${SERVER_URL}/v1/auth`, {
        publicKey: Buffer.from(accountKeypair.publicKey).toString('base64'),
        challenge,
        signature,
    });

    return {
        publicKey: accountKeypair.publicKey,
        secretKey: accountKeypair.secretKey,
        token: authResponse.data.token,
    };
}

/**
 * Clean up: delete the test account's data.
 * Calls server endpoints to remove sessions and machines.
 */
export async function cleanupTestAccount(account: TestAccount): Promise<void> {
    const headers = { Authorization: `Bearer ${account.token}` };

    try {
        // Delete all sessions
        const sessions = await axios.get(`${SERVER_URL}/v1/sessions`, { headers });
        for (const session of sessions.data) {
            await axios.delete(`${SERVER_URL}/v1/sessions/${session.id}`, { headers });
        }
    } catch {
        // Best effort cleanup
    }
}

export { SERVER_URL };
```

```typescript
// packages/idle-e2e/helpers/cli.ts
import { spawn, ChildProcess } from 'child_process';
import { resolve } from 'path';

const CLI_PATH = resolve(__dirname, '../../idle-cli');

export interface CLIProcess {
    process: ChildProcess;
    stdout: string[];
    stderr: string[];
    waitForOutput(pattern: string | RegExp, timeoutMs?: number): Promise<string>;
    kill(): void;
}

/**
 * Spawn the idle CLI as a child process for E2E testing.
 * Captures stdout/stderr and provides helpers for waiting on output.
 */
export function spawnCLI(args: string[], env?: Record<string, string>): CLIProcess {
    const proc = spawn('node', [
        '--require', 'tsx/cjs',
        resolve(CLI_PATH, 'src/index.ts'),
        ...args,
    ], {
        cwd: CLI_PATH,
        env: {
            ...process.env,
            NODE_NO_WARNINGS: '1',
            ...env,
        },
        stdio: ['pipe', 'pipe', 'pipe'],
    });

    const stdout: string[] = [];
    const stderr: string[] = [];

    proc.stdout?.on('data', (data) => stdout.push(data.toString()));
    proc.stderr?.on('data', (data) => stderr.push(data.toString()));

    return {
        process: proc,
        stdout,
        stderr,
        async waitForOutput(pattern: string | RegExp, timeoutMs = 30000): Promise<string> {
            const start = Date.now();
            while (Date.now() - start < timeoutMs) {
                const allOutput = stdout.join('');
                if (typeof pattern === 'string' && allOutput.includes(pattern)) return allOutput;
                if (pattern instanceof RegExp && pattern.test(allOutput)) return allOutput;
                await new Promise(r => setTimeout(r, 200));
            }
            throw new Error(`Timeout waiting for CLI output matching: ${pattern}\nGot: ${stdout.join('')}`);
        },
        kill() {
            proc.kill('SIGTERM');
        },
    };
}
```

```typescript
import { test, expect } from '@playwright/test';
import { createTestAccount, cleanupTestAccount, TestAccount, SERVER_URL } from '../helpers/auth';
import axios from 'axios';
import tweetnacl from 'tweetnacl';

let account: TestAccount;

test.describe('Authentication E2E', () => {
    test.beforeAll(async () => {
        account = await createTestAccount();
    });

    test.afterAll(async () => {
        await cleanupTestAccount(account);
    });

    test('account creation returns valid token', async () => {
        expect(account.token).toBeTruthy();
        expect(typeof account.token).toBe('string');
        expect(account.token.length).toBeGreaterThan(10);
    });

    test('token authenticates against server', async () => {
        const response = await axios.get(`${SERVER_URL}/v1/sessions`, {
            headers: { Authorization: `Bearer ${account.token}` },
        });
        expect(response.status).toBe(200);
    });

    test('terminal auth request flow works', async () => {
        // Simulate CLI: generate keypair and create auth request
        const terminalKeypair = tweetnacl.box.keyPair();
        const publicKeyBase64 = Buffer.from(terminalKeypair.publicKey).toString('base64');

        const reqResponse = await axios.post(`${SERVER_URL}/v1/auth/request`, {
            publicKey: publicKeyBase64,
            supportsV2: true,
        });

        expect(reqResponse.status).toBe(200);
```

### Task 11: ADR-002 — PGlite as Embedded Database

**Files:**

**Step 1: Write ADR-002**

Use the template from `~/Projects/.templates/adr/000-template.md`. Key content:

- **Context:** Server needs a database. Options: managed Postgres, SQLite, PGlite. VPS is a $5 IONOS box — running a separate Postgres process adds memory pressure and ops burden.
- **Decision:** Use PGlite (embedded Postgres via WASM). Zero-ops, single-process, Prisma-compatible.

### Task 13: ADR-004 — E2E Encryption Model

**Files:**
- Create: `docs/adr/004-e2e-encryption-model.md`

**Step 1: Write ADR-004**

- **Context:** Messages between CLI and web app transit through server. Server should not be able to read content (zero-knowledge relay).
- **Decision:** AES-256-GCM for session data encryption. TweetNaCl sealed boxes for key exchange during auth. Server stores encrypted blobs, cannot decrypt.
- **Consequences:** Strong privacy guarantees. Server metadata (session names, timestamps) still visible — known gap documented in SECURITY.md.
- **Alternatives:** Signal Protocol (overkill for 1:1 device pairing), no encryption (unacceptable for terminal content), TLS-only (server can read).
