```typescript
import { describe, it, expect } from 'vitest';

type Session = { id: string; machineId: string | null; tag: string };

function groupByMachine(sessions: Session[]): Map<string | null, Session[]> {
  const groups = new Map<string | null, Session[]>();
  for (const session of sessions) {
    const key = session.machineId;
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key)!.push(session);
  }
  return groups;
}

describe('Machine grouping', () => {
  const sessions: Session[] = [
    { id: '1', machineId: 'mac-1', tag: 'project-a' },
    { id: '2', machineId: 'mac-1', tag: 'project-b' },
    { id: '3', machineId: 'linux-1', tag: 'project-c' },
    { id: '4', machineId: null, tag: 'orphan-session' },
  ];

  it('groups sessions by machineId', () => {
    const groups = groupByMachine(sessions);
    expect(groups.get('mac-1')?.length).toBe(2);
    expect(groups.get('linux-1')?.length).toBe(1);
    expect(groups.get(null)?.length).toBe(1);
  });

  it('handles empty session list', () => {
    expect(groupByMachine([]).size).toBe(0);
  });

  it('handles all sessions from same machine', () => {
    const sameMachine = sessions.filter((s) => s.machineId === 'mac-1');
    const groups = groupByMachine(sameMachine);
    expect(groups.size).toBe(1);
    expect(groups.get('mac-1')?.length).toBe(2);
  });
});
```
