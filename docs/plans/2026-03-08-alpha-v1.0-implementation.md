import axios, { AxiosError } from 'axios';
import tweetnacl from 'tweetnacl';
import { createTestAccount, cleanupTestAccount, SERVER_URL, type TestAccount } from '../../../../idle-e2e/helpers/auth';

describe('Auth endpoint security', () => {

  afterAll(async () => { await cleanupTestAccount(account); });

  it('rejects truncated auth token', async () => {

        headers: { Authorization: `Bearer ${account.token.slice(0, 10)}` },
      });
      expect.fail('Should have thrown');
    } catch (e) {
      expect((e as AxiosError).response?.status).toBe(401);
    }
  });

  it('rejects empty bearer token', async () => {
    try {
      await axios.get(`${SERVER_URL}/v1/sessions`, {
        headers: { Authorization: 'Bearer ' },
      });
      expect.fail('Should have thrown');
    } catch (e) {
      expect((e as AxiosError).response?.status).toBe(401);
    }
  });

  it('rejects request with no auth header', async () => {
    try {
      await axios.get(`${SERVER_URL}/v1/sessions`);
      expect.fail('Should have thrown');
    } catch (e) {
      expect((e as AxiosError).response?.status).toBe(401);
    }
  });

  it('rejects malformed JSON in auth body', async () => {
    try {
      await axios.post(`${SERVER_URL}/v1/auth`, 'not-json', {
        headers: { 'Content-Type': 'application/json' },
      });
      expect.fail('Should have thrown');
    } catch (e) {
      expect([400, 415, 422]).toContain((e as AxiosError).response?.status);
    }
  });

  it('rejects auth with mismatched signature', async () => {
    const keypair = tweetnacl.sign.keyPair();
    const challenge = Buffer.from('test-challenge').toString('base64');
    const otherKeypair = tweetnacl.sign.keyPair();
    const signature = Buffer.from(
      tweetnacl.sign.detached(Buffer.from(challenge, 'base64'), otherKeypair.secretKey)
    ).toString('base64');

    try {
      await axios.post(`${SERVER_URL}/v1/auth`, {
        publicKey: Buffer.from(keypair.publicKey).toString('base64'),
        challenge,
        signature,
      });
      expect.fail('Should have thrown');
    } catch (e) {
      expect([400, 401]).toContain((e as AxiosError).response?.status);
    }
  });
});

import { describe, it, expect } from 'vitest';

type Session = { id: string; machineId: string | null; tag: string };

function groupByMachine(sessions: Session[]): Map<string | null, Session[]> {
  const groups = new Map<string | null, Session[]>();
  for (const session of sessions) {
    const key = session.machineId;
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key)!.push(session);
  }
  return groups;
}

describe('Machine grouping', () => {
  const sessions: Session[] = [
    { id: '1', machineId: 'mac-1', tag: 'project-a' },
    { id: '2', machineId: 'mac-1', tag: 'project-b' },
    { id: '3', machineId: 'linux-1', tag: 'project-c' },
    { id: '4', machineId: null, tag: 'orphan-session' },
  ];

  it('groups sessions by machineId', () => {
    const groups = groupByMachine(sessions);
    expect(groups.get('mac-1')?.length).toBe(2);
    expect(groups.get('linux-1')?.length).toBe(1);
    expect(groups.get(null)?.length).toBe(1);
  });

  it('handles empty session list', () => {
    expect(groupByMachine([]).size).toBe(0);
  });

  it('handles all sessions from same machine', () => {
    const sameMachine = sessions.filter((s) => s.machineId === 'mac-1');
    const groups = groupByMachine(sameMachine);
    expect(groups.size).toBe(1);
    expect(groups.get('mac-1')?.length).toBe(2);
  });
});
